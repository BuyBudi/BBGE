server/services/platformExtractors/ebayExtractor.js

- .x-price-primary

- AU $12,345
- A$12,345
- AUD $12,345

- postage
- shipping
- delivery
- returns
- ebay
- sponsored

- "convertedCurrentPrice"
- "currentPrice"
- "price"
- "value"
